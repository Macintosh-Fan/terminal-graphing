# Graphing
A simple C++ terminal graphing library made by Macintosh_Fan.

### Example 1
Input:
```cpp
#include <iostream>

#include "graphing.hpp"

int main() {
    // Creating the graph (f, printWidth, printHeight, minX, maxX, minY, maxY)
    Graphing::Graph graph([](double x) {return std::sin(x)/(x);}, 149, 8, -8 * M_PI, 8 * M_PI, -1, 1);

    // Printing the graph
    std::cout << graph.getResult();
    
    return 0;
}
```

### Example 2
Input:
```cpp
#include <iostream>

#include "graphing.hpp"

// A piecewise function
double f(double x) {
    if (x >= 1) {
        return cos(x);
    } else if (x > -1) {
        return -x;
    } else {
        return (x + 3) * (x + 3);
    }
}

int main() {
    // Creating the graph (f, printWidth, printHeight, minX, maxX, minY, maxY)
    Graphing::Graph graph(f, 149, 16, -4, 4 * M_PI, -1, 2);

    // Overriding default settings
    graph.emptySquare = " ";
    graph.fillSquare = "•";

    // Printing the graph
    std::cout << graph.getResult();
    
    return 0;
}
```
